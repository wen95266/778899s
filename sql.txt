CREATE TABLE IF NOT EXISTS lottery_results (
    id INT AUTO_INCREMENT PRIMARY KEY,
    issue VARCHAR(20) NOT NULL UNIQUE COMMENT '期号',
    open_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '开奖时间',
    numbers JSON NOT NULL COMMENT '平码数组 [1,2,3,4,5,6]',
    special_code INT NOT NULL COMMENT '特码',
    shengxiao VARCHAR(100) DEFAULT NULL COMMENT '生肖/波色描述',
    next_prediction JSON DEFAULT NULL COMMENT '下期预测号码数组',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
